Type

Example
// Get array of points along cubic curve

var start = {x: 0, y: 0};
var handle1 = {x: 3, y: 5};
var handle2 = {x: 7, y: -5};
var end = {x: 10, y: 0};
var increments = 1 / 10;
var points = [];
var time = 0;
while (time <= 1) {
  points.push({
    x: DL_.getPointOnCubicCurve(start.x, handle1.x, handle2.x, end.x, time),
    y: DL_.getPointOnCubicCurve(start.y, handle1.y, handle2.y, end.y, time)
  });
  time += increments;
}
